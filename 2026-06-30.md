## LeetCode 1358. Number of strings containing all three characters
Yeah I completed it without the help of hints or editorial, maybe (though I use LLM to debug, though the code is to te mo ugly)
### Approach
+ Prefix sum
+ Sliding window
+ Two pointer
### Code
```cpp
class Solution {
  public: int numberOfSubstrings(string s) {
    int n = s.size();

    vector < int > p1(n + 1, 0);
    vector < int > p2(n + 1, 0);
    vector < int > p3(n + 1, 0);

    for (int i = 1; i <= n; i++) {
      if (s[i - 1] == 'a') p1[i] = p1[i - 1] + 1, p2[i] = p2[i - 1], p3[i] = p3[i - 1];
      if (s[i - 1] == 'b') p2[i] = p2[i - 1] + 1, p1[i] = p1[i - 1], p3[i] = p3[i - 1];
      if (s[i - 1] == 'c') p3[i] = p3[i - 1] + 1, p2[i] = p2[i - 1], p1[i] = p1[i - 1];
    }

    int res = 0;

    int l = 1, r = 3;

    while (r <= n) {
      int cnt_a = p1[r] - p1[l - 1];
      int cnt_b = p2[r] - p2[l - 1];
      int cnt_c = p3[r] - p3[l - 1];
      if (cnt_a >= 1 && cnt_b >= 1 && cnt_c >= 1) {
        res += n - r + 1;
        l++;
      } else {
        r++;
      }
    }

    return res;
  }
};
```

### cool solution I want to attach here
It focus on the valid string **ends** at what position, and also trace the last position a character appears, so cool:)
```cpp
class Solution {
  public: int numberOfSubstrings(string s) {
    int n = s.size();

    vector < int > lastpos(3, -1);

    int res = 0;
    for (int i = 0; i < n; i++) {
      lastpos[s[i] - 'a'] = i;
      res += 1 + min(lastpos[0], min(lastpos[1], lastpos[2]));
    }

    return res;
  }
};
```
